// ═══════════════════════════════════════════════════ // FORGETAROUTE — Edge Function : delete-account // Supprime toutes les données d'un utilisateur // + son compte auth → réinscription repart de zéro // ═══════════════════════════════════════════════════ import { createClient } from 'https://esm.sh/@supabase/supabase-js@2' const corsHeaders = { 'Access-Control-Allow-Origin': '*', 'Access-Control-Allow-Headers': 'authorization, x-client-info, apikey, content-type', } Deno.serve(async (req) => { // Gestion CORS preflight if (req.method === 'OPTIONS') { return new Response('ok', { headers: corsHeaders }) } try { // ── 1. Récupérer le token de l'utilisateur connecté ── const authHeader = req.headers.get('Authorization') if (!authHeader) { return new Response( JSON.stringify({ error: 'Token manquant' }), { status: 401, headers: { ...corsHeaders, 'Content-Type': 'application/json' } } ) } // ── 2. Client avec le token utilisateur (pour vérifier l'identité) ── const supabaseUser = createClient( Deno.env.get('SUPABASE_URL') ?? '', Deno.env.get('SUPABASE_ANON_KEY') ?? '', { global: { headers: { Authorization: authHeader } } } ) // ── 3. Récupérer l'utilisateur connecté ── const { data: { user }, error: userError } = await supabaseUser.auth.getUser() if (userError || !user) { return new Response( JSON.stringify({ error: 'Utilisateur non authentifié' }), { status: 401, headers: { ...corsHeaders, 'Content-Type': 'application/json' } } ) } const userId = user.id // ── 4. Client admin (service_role) pour supprimer les données ── const supabaseAdmin = createClient( Deno.env.get('SUPABASE_URL') ?? '', Deno.env.get('SUPABASE_SERVICE_ROLE_KEY') ?? '' ) // ── 5. Récupérer le stripe_subscription_id avant suppression ── const { data: profil } = await supabaseAdmin .from('profils') .select('stripe_subscription_id, abonnement') .eq('id', userId) .single() // ── 6. Annuler l'abonnement Stripe si actif ── if (profil?.stripe_subscription_id && profil?.abonnement === 'premium') { try { const stripeKey = Deno.env.get('STRIPE_SECRET_KEY') ?? '' const stripeResp = await fetch( `https://api.stripe.com/v1/subscriptions/${profil.stripe_subscription_id}`, { method: 'DELETE', headers: { 'Authorization': `Bearer ${stripeKey}`, 'Content-Type': 'application/x-www-form-urlencoded', } } ) const stripeData = await stripeResp.json() if (stripeData.error) { console.error('Erreur annulation Stripe:', stripeData.error) // On continue quand même — ne pas bloquer la suppression } else { console.log('Abonnement Stripe annulé:', profil.stripe_subscription_id) } } catch (stripeErr) { console.error('Erreur Stripe inattendue:', stripeErr) } } // ── 7. Supprimer toutes les données dans l'ordre ── // (les FK doivent être supprimées avant le profil et le compte auth) // 7a. Supprimer les réponses const { error: reponsesError } = await supabaseAdmin .from('reponses') .delete() .eq('user_id', userId) if (reponsesError) { console.error('Erreur suppression reponses:', reponsesError) // On continue quand même — ne pas bloquer si la table est vide } // 7b. Supprimer les sessions d'entraînement const { error: sessionsError } = await supabaseAdmin .from('sessions') .delete() .eq('user_id', userId) if (sessionsError) { console.error('Erreur suppression sessions:', sessionsError) } // 7c. Supprimer les examens blancs const { error: examensError } = await supabaseAdmin .from('examens') .delete() .eq('user_id', userId) if (examensError) { console.error('Erreur suppression examens:', examensError) } // 7d. Supprimer le profil const { error: profilError } = await supabaseAdmin .from('profils') .delete() .eq('id', userId) if (profilError) { console.error('Erreur suppression profil:', profilError) } // 7e. Supprimer le compte auth (doit être en dernier) const { error: authError } = await supabaseAdmin.auth.admin.deleteUser(userId) if (authError) { console.error('Erreur suppression compte auth:', authError) return new Response( JSON.stringify({ error: 'Erreur lors de la suppression du compte : ' + authError.message }), { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } } ) } // ── 6. Succès — toutes les données sont supprimées ── console.log(`Compte supprimé avec succès : ${userId}`) return new Response( JSON.stringify({ success: true, message: 'Compte et toutes les données supprimés.' }), { status: 200, headers: { ...corsHeaders, 'Content-Type': 'application/json' } } ) } catch (err) { console.error('Erreur inattendue:', err) return new Response( JSON.stringify({ error: 'Erreur serveur inattendue' }), { status: 500, headers: { ...corsHeaders, 'Content-Type': 'application/json' } } ) } })
